# Revenue by customer state (geographic MDX) @router.get("/analytics/mdx/revenue-by-segment") def revenue_by_segment(): with engine.connect() as conn: result = conn.execute(text(""" SELECT c.customer_state, COUNT(DISTINCT o.order_id) as orders, SUM(oi.price) as revenue FROM orders o JOIN customers c ON o.customer_id = c.customer_id JOIN order_items oi ON o.order_id = oi.order_id WHERE o.order_status = 'delivered' GROUP BY c.customer_state ORDER BY revenue DESC LIMIT 15 """)) return [dict(zip(result.keys(), r)) for r in result.fetchall()] # Churn rate by demographics (pivot table) @router.get("/analytics/mdx/churn-by-demographics") def churn_by_demographics(): with engine.connect() as conn: result = conn.execute(text(""" SELECT "Gender", "PreferredLoginDevice", "CityTier", COUNT(*) as total, SUM("Churn") as churned, ROUND(AVG("Churn_Probability")::numeric, 3) as avg_churn_prob FROM customer_churn GROUP BY "Gender", "PreferredLoginDevice", "CityTier" ORDER BY avg_churn_prob DESC """)) return [dict(zip(result.keys(), r)) for r in result.fetchall()]
