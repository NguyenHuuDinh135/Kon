3. FastAPI cơ bản

FastAPI tạo API endpoint cực đơn giản:

from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy import text
from db_core import engine

router = APIRouter()

# GET endpoint — Lấy danh sách
@router.get("/products")
def get_products(search: str = None):
    with engine.connect() as conn:
        if search:
            result = conn.execute(text(
                "SELECT * FROM products WHERE product_category_name ILIKE :s LIMIT 50"
            ), {"s": f"%{search}%"})
        else:
            result = conn.execute(text("SELECT * FROM products LIMIT 50"))
        rows = result.fetchall()
        return [dict(zip(result.keys(), row)) for row in rows]

# POST endpoint — Tạo mới
@router.post("/orders")
def create_order(data: dict):
    # Validate input
    if not data.get("product_id"):
        raise HTTPException(status_code=400, detail="product_id required")
    # Insert to database
    with engine.connect() as conn:
        conn.execute(text(
            "INSERT INTO orders (product_id, quantity) VALUES (:pid, :qty)"
        ), {"pid": data["product_id"], "qty": data.get("quantity", 1)})
        conn.commit()
    return {"message": "Order created"}

# GET /products — Lấy danh sách sản phẩm
@router.get("/products")
def get_products(search: str = None, limit: int = 50):
    with engine.connect() as conn:
        query = """
            SELECT product_id, product_category_name,
                   product_weight_g, product_length_cm
            FROM products
        """
        if search:
            query += " WHERE product_category_name ILIKE :s"
            result = conn.execute(text(query + " LIMIT :lim"),
                                  {"s": f"%{search}%", "lim": limit})
        else:
            result = conn.execute(text(query + " LIMIT :lim"), {"lim": limit})
        return [dict(zip(result.keys(), row)) for row in result.fetchall()]

4.2. Orders CRUD

# GET /orders — Lấy danh sách đơn hàng
@router.get("/orders")
def get_orders(limit: int = 50):
    with engine.connect() as conn:
        result = conn.execute(text("""
            SELECT o.order_id, o.customer_id, o.order_status,
                   o.order_purchase_timestamp,
                   SUM(oi.price + oi.freight_value) as total
            FROM orders o
            LEFT JOIN order_items oi ON o.order_id = oi.order_id
            GROUP BY o.order_id, o.customer_id, o.order_status,
                     o.order_purchase_timestamp
            ORDER BY o.order_purchase_timestamp DESC
            LIMIT :lim
        """), {"lim": limit})
        return [dict(zip(result.keys(), row)) for row in result.fetchall()]

4.3. Notifications

# GET /notifications — Danh sách thông báo
@router.get("/notifications")
def get_notifications():
    with engine.connect() as conn:
        result = conn.execute(text("""
            SELECT id, title, message, read, created_at
            FROM notifications
            ORDER BY created_at DESC
            LIMIT 50
        """))
        return [dict(zip(result.keys(), row)) for row in result.fetchall()]

# GET /notifications/unread-count
@router.get("/notifications/unread-count")
def unread_count():
    with engine.connect() as conn:
        result = conn.execute(text(
            "SELECT COUNT(*) FROM notifications WHERE read = false"
        ))
        return {"count": result.scalar()}

# PUT /notifications/read-all — Đánh dấu tất cả đã đọc
@router.put("/notifications/read-all")
def mark_all_read():
    with engine.connect() as conn:
        conn.execute(text("UPDATE notifications SET read = true WHERE read = false"))
        conn.commit()
    return {"message": "All marked as read"}

6. Viết Tests (pytest)

apps/api/tests/test_api.py
import pytest
from fastapi.testclient import TestClient
from apps.api.main import app

client = TestClient(app)

# Test lấy danh sách sản phẩm
def test_get_products():
    response = client.get("/products")
    assert response.status_code == 200
    data = response.json()
    assert isinstance(data, list)

# Test lấy đơn hàng (cần auth)
def test_get_orders_unauthorized():
    response = client.get("/orders")
    assert response.status_code == 401  # Chưa login

def test_get_orders_authorized():
    # Login trước
    login = client.post("/auth/login", json={
        "username": "admin", "password": "admin123"
    })
    token = login.json()["access_token"]

    # Gọi API với token
    response = client.get("/orders", headers={
        "Authorization": f"Bearer {token}"
    })
    assert response.status_code == 200

# Test health check
def test_health():
    response = client.get("/health")
    assert response.status_code == 200
    assert response.json()["status"] == "healthy"

# Test notifications
def test_get_notifications():
    login = client.post("/auth/login", json={
        "username": "admin", "password": "admin123"
    })
    token = login.json()["access_token"]
    response = client.get("/notifications", headers={
        "Authorization": f"Bearer {token}"
    })
    assert response.status_code == 200

8. SQL cơ bản cần biết

-- Lấy 10 đơn hàng gần nhất
SELECT * FROM orders ORDER BY order_purchase_timestamp DESC LIMIT 10;

-- Đếm đơn hàng theo trạng thái
SELECT order_status, COUNT(*) FROM orders GROUP BY order_status;

-- Join orders + items để tính tổng
SELECT o.order_id, SUM(oi.price) as total
FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
GROUP BY o.order_id;

-- Tìm khách hàng có churn risk cao
SELECT "CustomerID", "Churn_Probability"
FROM customer_churn
WHERE "Churn_Probability" > 0.7
ORDER BY "Churn_Probability" DESC;
