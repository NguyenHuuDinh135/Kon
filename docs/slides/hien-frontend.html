4. Cách gọi API từ Frontend

apps/web/lib/api.ts
// Hàm gọi API có sẵn, bạn chỉ cần import và dùng:
import { apiClient } from "@/lib/api";

// Lấy danh sách sản phẩm
const products = await apiClient("/products");

// Lấy danh sách đơn hàng (cần auth)
const orders = await apiClient("/orders", {
  headers: { Authorization: `Bearer ${token}` }
});

// Tạo đơn hàng mới
const newOrder = await apiClient("/orders", {
  method: "POST",
  body: JSON.stringify({ product_id: 1, quantity: 2 })
});
Lưu ý: Token lưu trong localStorage key kon_token. Khi user login thành công, token được lưu tự động.

6. Trang bạn cần làm (chi tiết)

6.1. Trang chủ Storefront

app/(storefront)/page.tsx

Hiển thị: Hero banner, sản phẩm nổi bật, danh mục, khuyến mãi.

6.2. Trang sản phẩm

app/(storefront)/products/page.tsx

Gọi API: GET /products → Hiển thị grid cards, có search + pagination.

6.3. Trang đăng nhập

app/(storefront)/login/page.tsx

Form: username + password → Gọi POST /auth/login → Lưu token → Redirect dashboard.

6.4. Trang đăng ký

app/(storefront)/register/page.tsx

Form: username + email + password → Gọi POST /auth/register.

6.5. Dashboard Orders

app/(admin)/dashboard/orders/page.tsx

Bảng hiển thị đơn hàng: order_id, customer, total, status, date. Gọi GET /orders.

6.6. Dashboard Customers

app/(admin)/dashboard/customers/page.tsx

Bảng hiển thị khách hàng: ID, name, email, orders count, churn risk. Gọi GET /customers.

9. Mẫu trang hoàn chỉnh

"use client";

import { useEffect, useState } from "react";
import { Card, CardContent, CardHeader, CardTitle } from "@workspace/ui/components/card";
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@workspace/ui/components/table";
import { Badge } from "@workspace/ui/components/badge";

export default function OrdersPage() {
  const [orders, setOrders] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const token = localStorage.getItem("kon_token");
    fetch("http://localhost:8000/orders", {
      headers: { Authorization: `Bearer ${token}` }
    })
      .then(res => res.json())
      .then(data => { setOrders(data); setLoading(false); })
      .catch(() => setLoading(false));
  }, []);

  if (loading) return <div className="p-6">Đang tải...</div>;

  return (
    <div className="p-6 space-y-6">
      <h1 className="text-2xl font-bold">Quản lý đơn hàng</h1>
      <Card>
        <CardContent className="p-0">
          <Table>
            <TableHeader>
              <TableRow>
                <TableHead>Mã đơn</TableHead>
                <TableHead>Khách hàng</TableHead>
                <TableHead>Tổng tiền</TableHead>
                <TableHead>Trạng thái</TableHead>
              </TableRow>
            </TableHeader>
            <TableBody>
              {orders.map((order) => (
                <TableRow key={order.order_id}>
                  <TableCell>{order.order_id}</TableCell>
                  <TableCell>{order.customer_id}</TableCell>
                  <TableCell>${order.total}</TableCell>
                  <TableCell>
                    <Badge variant={order.status === 'delivered' ? 'default' : 'secondary'}>
                      {order.status}
                    </Badge>
                  </TableCell>
                </TableRow>
              ))}
            </TableBody>
          </Table>
        </CardContent>
      </Card>
    </div>
  );
}
