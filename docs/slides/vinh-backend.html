3. Code mẫu: Customers API

apps/api/routers/ (thêm vào file hiện có)
from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy import text
from db_core import engine

router = APIRouter()

# GET /customers — Danh sách khách hàng (từ bảng customer_churn)
@router.get("/customers")
def get_customers(
    limit: int = Query(50, le=200),
    risk: str = Query(None, description="Filter: high, medium, low")
):
    with engine.connect() as conn:
        query = """
            SELECT "CustomerID", "Gender", "Tenure", "SatisfactionScore",
                   "OrderCount", "DaySinceLastOrder", "CashbackAmount",
                   "PreferedOrderCat", "Churn_Probability", "Cluster"
            FROM customer_churn
            WHERE "Churn_Probability" IS NOT NULL
        """
        params = {"lim": limit}

        if risk == "high":
            query += ' AND "Churn_Probability" > 0.7'
        elif risk == "medium":
            query += ' AND "Churn_Probability" BETWEEN 0.3 AND 0.7'
        elif risk == "low":
            query += ' AND "Churn_Probability" < 0.3'

        query += ' ORDER BY "Churn_Probability" DESC LIMIT :lim'
        result = conn.execute(text(query), params)
        rows = result.fetchall()
        return [dict(zip(result.keys(), row)) for row in rows]


# GET /customers/:id — Chi tiết 1 khách hàng
@router.get("/customers/{customer_id}")
def get_customer(customer_id: int):
    with engine.connect() as conn:
        result = conn.execute(text("""
            SELECT * FROM customer_churn WHERE "CustomerID" = :cid
        """), {"cid": customer_id})
        row = result.fetchone()
        if not row:
            raise HTTPException(status_code=404, detail="Customer not found")
        return dict(zip(result.keys(), row))

4. Code mẫu: Campaigns API

from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy import text
from db_core import engine
from datetime import datetime

router = APIRouter()

# GET /campaigns — Danh sách chiến dịch
@router.get("/campaigns")
def get_campaigns():
    with engine.connect() as conn:
        result = conn.execute(text("""
            SELECT id, name, segment, status, channel, message, created_at
            FROM campaigns
            ORDER BY created_at DESC
        """))
        return [dict(zip(result.keys(), row)) for row in result.fetchall()]


# POST /campaigns — Tạo chiến dịch mới
@router.post("/campaigns")
def create_campaign(data: dict):
    name = data.get("name")
    segment = data.get("segment", "general")
    channel = data.get("channel", "email")
    message = data.get("message", "")

    if not name:
        raise HTTPException(status_code=400, detail="Campaign name required")

    with engine.connect() as conn:
        conn.execute(text("""
            INSERT INTO campaigns (name, segment, status, channel, message, created_at)
            VALUES (:name, :segment, 'draft', :channel, :message, :now)
        """), {
            "name": name,
            "segment": segment,
            "channel": channel,
            "message": message,
            "now": datetime.utcnow()
        })
        conn.commit()
    return {"message": f"Campaign '{name}' created as draft"}


# PUT /campaigns/:id/approve — Duyệt chiến dịch
@router.put("/campaigns/{campaign_id}/approve")
def approve_campaign(campaign_id: int):
    with engine.connect() as conn:
        result = conn.execute(text(
            "UPDATE campaigns SET status = 'approved' WHERE id = :id AND status = 'draft'"
        ), {"id": campaign_id})
        conn.commit()
        if result.rowcount == 0:
            raise HTTPException(status_code=404, detail="Campaign not found or already approved")
    return {"message": "Campaign approved"}


# PUT /campaigns/:id/execute — Thực thi chiến dịch
@router.put("/campaigns/{campaign_id}/execute")
def execute_campaign(campaign_id: int):
    with engine.connect() as conn:
        result = conn.execute(text(
            "UPDATE campaigns SET status = 'executed' WHERE id = :id AND status = 'approved'"
        ), {"id": campaign_id})
        conn.commit()
        if result.rowcount == 0:
            raise HTTPException(status_code=404, detail="Campaign not found or not approved")
    return {"message": "Campaign executed"}

5. Auth — Cách protect endpoint

API sử dụng JWT token. Cách bảo vệ endpoint chỉ admin mới truy cập được:

from apps.api.routers.auth import get_current_user, require_admin

# Endpoint cần đăng nhập
@router.get("/customers")
def get_customers(current_user = Depends(get_current_user)):
    # current_user chứa thông tin user đã login
    ...

# Endpoint chỉ admin
@router.post("/campaigns")
def create_campaign(data: dict, admin = Depends(require_admin)):
    # Chỉ user có role=admin mới vào được
    ...
Flow auth:
1. User gọi POST /auth/login với username + password
2. Server trả về JWT token
3. Mọi request sau đều gửi header: Authorization: Bearer <token>
4. Server decode token → biết user nào đang gọi

CREATE TABLE campaigns (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    segment VARCHAR(100),         -- 'high_churn', 'vip', 'at_risk'...
    status VARCHAR(50) DEFAULT 'draft',  -- draft → approved → executed
    channel VARCHAR(100),         -- 'email', 'push', 'sms'
    message TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

7. Viết Tests cho API

apps/api/tests/test_customers.py
import pytest
from fastapi.testclient import TestClient
from apps.api.main import app

client = TestClient(app)

def get_admin_token():
    """Helper: lấy token admin"""
    resp = client.post("/auth/login", json={
        "username": "admin", "password": "admin123"
    })
    return resp.json()["access_token"]

def test_get_customers():
    token = get_admin_token()
    resp = client.get("/customers", headers={
        "Authorization": f"Bearer {token}"
    })
    assert resp.status_code == 200
    data = resp.json()
    assert isinstance(data, list)
    assert len(data) > 0

def test_get_customers_filter_high_risk():
    token = get_admin_token()
    resp = client.get("/customers?risk=high", headers={
        "Authorization": f"Bearer {token}"
    })
    assert resp.status_code == 200
    for customer in resp.json():
        assert customer["Churn_Probability"] > 0.7

def test_get_customer_by_id():
    token = get_admin_token()
    resp = client.get("/customers/100", headers={
        "Authorization": f"Bearer {token}"
    })
    assert resp.status_code == 200
    assert resp.json()["CustomerID"] == 100

def test_get_customer_not_found():
    token = get_admin_token()
    resp = client.get("/customers/99999", headers={
        "Authorization": f"Bearer {token}"
    })
    assert resp.status_code == 404

def test_create_campaign():
    token = get_admin_token()
    resp = client.post("/campaigns", json={
        "name": "Test Campaign",
        "segment": "high_churn",
        "channel": "email",
        "message": "Giảm 20% cho đơn hàng tiếp theo!"
    }, headers={"Authorization": f"Bearer {token}"})
    assert resp.status_code == 200

def test_get_campaigns():
    token = get_admin_token()
    resp = client.get("/campaigns", headers={
        "Authorization": f"Bearer {token}"
    })
    assert resp.status_code == 200
    assert isinstance(resp.json(), list)
